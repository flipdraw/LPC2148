#include"lcd.h"
int i;

void start(unsigned char x)
{
I2CONSET=x;				//10 0000 START
while((I2CONSET&0X08)==0X00);
}

void idata(unsigned char x)
{
I2DAT=x;	   			//	   SLAVE ADD & data.
I2CONCLR=0X08;
while((I2CONSET&0X08)==0X00);
}

void stop(unsigned char x)
{
I2CONSET=x;			  	//STOP
I2CONCLR=0X08;
for(i=0;i<210;i++);
//I2CONCLR=0X08;

}
